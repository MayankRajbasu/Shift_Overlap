                const report = document.getElementById("report");

                const textContent = report.innerText;

                const blobHtml = new Blob([htmlContent], { type: "text/html" });
                const blobText = new Blob([textContent], { type: "text/plain" });

                const data = [
                    new ClipboardItem({
                        "text/html": blobHtml,

                    })

                    alert("Copy failed: " + err);
                });
